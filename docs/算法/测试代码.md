int rotate_array(const vector<int>& nums, int target) {
    if (nums.size() == 0) return -1;
    int lo = 0;
    int hi = nums.size()-1;
    // 找最小元素
    // 考虑情况：[10,1,10,10,10]，因此不能用nums[0]做哨兵
    while (lo < hi) {
        int m = lo + (hi-lo)/2;
        if (nums[m] > nums[hi]) lo = m+1;  
        else if (nums[m] < nums[hi]) hi = m;
        else --hi;
    }
	
    if (target > nums[0]) {
         // 在前半段
        return find(nums, 0, hi-1, target);
    } else {
        // 在后半段
        return find(nums, hi, nums.size()-1, target);
    }
}

int find(const vector<int>& nums, int i, int j, int target) {
    int lo = i;
    int hi = j;
    while (lo < hi) {
        int m = lo + (hi-lo)/2;
        if (target == nums[m]) return m;
        else if (target < nums[m]) hi = m-1;
        else lo = m+1;
    }
    return -1;
}

class PriorityQueue
{
    int* arr;
    int len;
public:    
    bool is_empty() {
        return len == 0;
    }
    
    int size() {
        return len;
    }
    
    void insert(int v) {
        arr[++len] = v;  // 上浮
        swin(len);
    }
    
    int del_max() {
        int max = arr[1]; // 
        // 交换第一个和最后一个元素，长度-1
        swap(arr, 1, len--);
        // 下沉
        sink(1);
        return max;
    }
    
    // 上浮
    void swin(int k) {
        while (k > 1 && arr[k] > arr[k/2]) {
            swap(arr, k/2, k);
            k /= 2;
        }
    }
    
    // 下沉
    void sink(int k) {
        while (len >= 2*k) {
            int j = 2*k;
            // 比较左和右，保留最大的与父结点比较
            if (arr[j] < arr[j+1]) j++;
            if (arr[k] > arr[j]) break;
            swap(arr, k, j);
            k = j; // 继续下沉
        }
    }
}


int main() {
    PriorityQueue q(3);
    q.insert(1);
    q.insert(5);
    q.insert(2);
    
    cout << q.del_max() << endl;  // 5
    cout << q.size() << endl; // size: 2
    cout << q.del_max() << endl;  // 2
    cout << q.del_max() << end;  // 1
    // 5 
    // 2
    // 1
}

// 长度为N的数组，输出第k大的元素, O(n)
// 无序，降序排序，输出第k个元素
// n
// min max
int partition(int * arr, int l, int h) {
    int index = l + rang() % (h -l +1);
    swap(arr, index, l);
    int v = arr[l];
    int i = l;
    int j = h;
    while (true) {
        while (arr[++i] < v) if (i >= h) break;
        while (arr[--j] > v) if (j <= l) break;
        if (i >= j) break;
        swap(arr, i, j);
    }
    swap(arr, l, j);
    return l;
}

int quickSelect(int * arr, int l, int h, index) {
    int q = partition(arr, l, h);
    if (q == index) { // 
        return arr[q];
    } else {
        if (q < index) {
            return quickSelect(arr, q+1, h, index);
        } else {
            return quickSelect(arr, l, q-1, index);
        }
    }
}

int solve(int * arr, int nums, int k) {
    return quickSelect(arr, 0, nums, k);
}
