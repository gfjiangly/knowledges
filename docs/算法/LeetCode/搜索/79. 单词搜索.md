## 题目描述

给定一个二维网格和一个单词，找出该单词是否存在于网格中。

单词必须按照字母顺序，通过相邻的单元格内的字母构成，其中“相邻”单元格是那些水平相邻或垂直相邻的单元格。同一个单元格内的字母不允许被重复使用。

示例:

```
board =
[
  ['A','B','C','E'],
  ['S','F','C','S'],
  ['A','D','E','E']
]

给定 word = "ABCCED", 返回 true.
给定 word = "SEE", 返回 true.
给定 word = "ABCB", 返回 false.
```

## 回溯法

```C++
class Solution {
    int d[4][2] = {{0, 1}, {1, 0}, {0, -1}, {-1, 0}};
    bool** flag;
    int h, w, size;
    bool backtrack(const vector<vector<char>>& board, const string& word, int i, int j, int k) {
        if (i < 0 || j < 0 || i >= h || j >= w || k >= word.size() || flag[i][j]) return false;
        //printf("i=%d, j=%d, v=%c, w=%c\n", i, j, board[i][j], word[k]);
        if (board[i][j] == word[k]) {
            if (k == word.size()-1) return true;
            else {
                for (int n = 0; n < 4; ++n) {
                    flag[i][j] = true;
                    if (backtrack(board, word, i+d[n][0], j+d[n][1], k+1)) return true;
                    flag[i][j] = false;
                }
            }
        }
        return false;
    }
public:
    bool exist(vector<vector<char>>& board, string word) {
        size = word.size();
        if (size == 0) return false;
        h = board.size();
        if (h == 0) return false;
        w = board[0].size();

        // 如果允许修改原数组，flag可省略
        flag = new bool*[h];
        for (int i = 0; i < h; ++i) {
            flag[i] = new bool[w];
            for (int j = 0; j < w; ++j) {
                flag[i][j] = false;
            }
        }

        for (int i = 0; i < h; ++i) {
            for (int j = 0; j < w; ++j) {
                if (board[i][j] == word[0]) {
                    if (backtrack(board, word, i, j, 0)) return true;
                }
            }
        }
        return false;
    }
};
```
