```bash
# 创建名为xxx的新分支
git checkout -b xxx
# checkout -b 是复合命令，一个是创建，一个是切换：
# 创建分支的命令是 git branch dev
# 切换分支的命令是 git checkout dev

# 查看当前所有的分支，结果显示带*号的，而且颜色是绿色的即为我们当前所在的分支
git branch -a

# 切换分支
# 从当前的master分支切换到online分支上面
git checkout xxx
# 此时可以查看分支, 使用git branch即可看到master和online分支
git branch

# 合并master分支到当前分支
git merge master

# 合并xxx分支到master（先切到master分支）
git merge xxx

# 删除本地xxx分支
git branch -d xxx

# 删除远程xxx分支
git push origin --delete xxx
```
